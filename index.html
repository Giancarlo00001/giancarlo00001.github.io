        // Sticky CTAs on scroll with footer detection

        window.addEventListener('scroll', function() {

            const scrollTop = window.pageYOffset || document.documentElement.scrollTop;

            const footer = document.querySelector('.footer');

            const footerTop = footer.getBoundingClientRect().top + window.pageYOffset;

            const windowHeight = window.innerHeight;

            

            // Desktop/Tablet sticky CTA (768px and up)

            if (window.innerWidth >= 768) {

                const desktopCTA = document.getElementById('stickyCtaDesktop');

                const ctaHeight = 60; // Approximate height of CTA button

                const bottomOffset = 30; // Original bottom offset

                

                if (scrollTop > 300) {

                    desktopCTA.classList.add('show');

                    

                    // Check if CTA would overlap footer

                    if (scrollTop + windowHeight > footerTop - ctaHeight - bottomOffset) {

                        const newBottom = scrollTop + windowHeight - footerTop + ctaHeight + bottomOffset;

                        desktopCTA.style.bottom = `${newBottom}px`;

                    } else {

                        desktopCTA.style.bottom = '30px';

                    }

                } else {

                    desktopCTA.classList.remove('show');

                }

            } 

            // Mobile sticky CTA (below 768px)

            else {

                const mobileCTA = document.getElementById('mobileStickyCTA');

                const mobileCtaHeight = mobileCTA.offsetHeight || 60;

                

                if (scrollTop > 300) {

                    // Check if we're near the footer

                    if (scrollTop + windowHeight > footerTop) {

                        mobileCTA.style.transform = 'translateY(100%)';

                    } else {

                        mobileCTA.classList.add('show');

                        mobileCTA.style.transform = 'translateY(0)';

                    }

                } else {

                    mobileCTA.classList.remove('show');

                }

            }

        });
